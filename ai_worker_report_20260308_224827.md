    if len(values) <= window:
        return None

    gains: List[float] = []
    losses: List[float] = []

    for i in range(1, len(values)):
        diff = values[i] - values[i - 1]
        if diff > 0:
            gains.append(diff)
            losses.append(0.0)
        else:
            gains.append(0.0)
            losses.append(-diff)

    if len(gains) < window:
        return None

    avg_gain = mean(gains[-window:])
    avg_loss = mean(losses[-window:])

        return 100.0

    rs = avg_gain / avg_loss
    return 100.0 - (100.0 / (1 + rs))

## Issue 2: ATR Period Check Off-by-One

**SEVERITY: MEDIUM**
**FILE: pre_mover_signal.py**
**FUNCTION: _atr, LINE: 50**
**ROOT CAUSE:** The check `len(closes) < period + 2` is inconsistent. For ATR calculation, you need `period` true range values, which requires `period + 1` price bars (since TR uses previous close). The `+ 2` appears to be overly conservative, potentially rejecting valid data.

```python
if len(closes) < period + 2:
    return None
```

For a 5-period ATR, you need 6 bars minimum (5 TR values from bars 1-5 using closes 0-4). The current check requires 7 bars for ATR(5), wasting one bar.

def extract_code_block(text: str) -> str:
    """Extract code from markdown code blocks or raw text."""
    # Match ```python ... ``` or ``` ... ```
    match = re.search(r'```(?:python)?\s*\n?(.*?)\n?```', text, re.DOTALL)
